NXT I2C Library

PFMate.controlMotor()

Description

Control the PF Motors

Syntax

controlMotor(uint8_t channel, uint8_t control, uint8_t operation, uint8_t speed);

Parameters

uint8_t channel: The channel the PFMate Reciever is set to.  PF_Channel_1, 2, 3, or 4

uint8_t control: Which motor to control. PF_Contol_Both, A, or B

uint8_t operation: The operation of motor(s). PF_Operation_Forward, Reverse, Float, or Brake


uint8_t speed: The speed you want the motor(s) to run at.  [0, inf] or Servo_Speed_Full

Returns

None

Example

#include <Wire.h>               // Needs to be included here for //
                                // the NXTI2CDevice to work      //
#include <NXTI2CDevice.h>
#include <PFMate.h>

PFMate          pf;

void
setup()
{
  Serial.begin(9600);         // start serial for output

  char            str[40];

  delay(1000);
  pf.controlMotor(PF_Channel_4,	          // PF_Channel_1, 2, 3, or 4
                  PF_Contol_Both,         // PF_Contol_Both, A, or B
                  PF_Operation_Forward,   // PF_Operation_Forward, Reverse, Float, or Brake
                  PF_Speed_Full);         // [1, 7] or PF_Speed_Full, Medium, or Slow
  pf.sendSignal();
  delay(1000);                  
  Serial.println("Program Finished");
}

void
loop()                          // After the setup the LED blinks to show
                                // the program is complete
{
  digitalWrite(13, HIGH);     // set the LED on
  delay(1000);                // wait 
  digitalWrite(13, LOW);      // set the LED off
  delay(1000);                // wait
}
