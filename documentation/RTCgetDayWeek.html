NXT I2C Library

RTC.getDayWeek()

Description

Get the current day of the week reading

Syntax

getSeconds();

Parameters

None

Returns

Day of the week (0-6) 1-Monday, 2-Tuesday, ... etc.

Example

#include <Wire.h>               // Needs to be included here for //
                                // the NXTI2CDevice to work      //
#include <NXTI2CDevice.h>
#include <RTC.h>

RTC     rtc;

void
setup()
{
    Serial.begin(9600);         // start serial for output
}
   
void
loop()
{
  char str[50]; 
    sprintf (str, 
    "Date & time: %.2d-%.2d-%.2d, %.2d:%.2d:%.2d", 
            rtc.getMonth(), rtc.getDayMonth(), rtc.getYear(),            
            rtc.getHours(), rtc.getMinutes(), rtc.getSeconds());
    Serial.println(str);

  delay(1000);
}
