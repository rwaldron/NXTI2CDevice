NXT I2C Library

DISTNx.getType()

Description

Gets the current sensor module type

Syntax

getType(); 

Parameters

None

Returns

Current sensor module type. 2 (Short Range), 3 (Medium Range) or 4 (Long Range)

Example

#include <Wire.h>               // Needs to be included here for //
                                // the NXTI2CDevice to work      //
#include <NXTI2CDevice.h>
#include <DISTNx.h>

DISTNx          dist;

void
setup()
{
    Serial.begin(9600);         // start serial for output

    Serial.print("Voltage:");
    Serial.println(dist.getVolt());
    Serial.print("Sensor Type:");
    Serial.println(dist.getType());

}

void
loop()
{
    delay(1000);
    Serial.print("Distance:");
    Serial.println(dist.getDist());

}
